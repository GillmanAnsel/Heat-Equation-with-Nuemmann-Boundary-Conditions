\section{Problem Statement}
Consider the one-dimensional Heat equation:
\begin{equation}
    \frac{\partial u}{\partial t} = \frac{\partial^{2}u}{\partial x^{2}} + f(x,t), \quad x \in (0,L), \quad t \in (0, T_{F}]
\end{equation}
subject to Neumann boundary conditions:
\begin{equation}
    u_{x}(0,t) = a, \quad u_{x}(L,t) = b
\end{equation}
and an initial condition:
\begin{equation}
    u(x,0) = u_{0}(x)
\end{equation}
where $a, b \in \mathbb{R}$, $L > 0$, and $T_{F} > 0$.

\section{Task 1}
Approximate the solution to this initial boundary value problem by following these steps:
\begin{enumerate}
    \item Use the \textbf{implicit Euler method} for time discretization.
    \item Derive the \textbf{weak formulation} by multiplying the equation by a test function $v(x)$ and integrating by parts.
    \item Construct the \textbf{finite element approximation scheme} using piecewise linear basis functions for discretizing the spatial domain.
    \item Formulate the \textbf{fully discrete scheme} and the resulting system of linear equations.
\end{enumerate}

\section{Task 2}
Implement the numerical scheme from step 4 in Task 1 using a programming language of your choice (MATLAB is recommended).

\section{Notation}
\begin{itemize}
    \item For a fixed $N \in \mathbb{N}$, let $t_{n} = n\tau$ $(n=0,1,...,N)$ be a uniform partition of $[0, T_{F}]$ with $\tau = \frac{T_{F}}{N}$.
    \item $u(t_{n})$ represents the exact solution of the PDE at the $n^{th}$ time step.
    \item $u_{h}^{n}$ represents the finite element approximation of $u(t_{n})$.
    \item $\|v\|_{L^{2}((0,1))}^{2} := \int_{0}^{1} |v|^{2} dx$.
\end{itemize}
